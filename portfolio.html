Moteur.h — C++ 


// Classe Moteur — contrôle un moteur DC via L298N
class Moteur {
  private:
    int pin_in1, pin_in2, pin_pwm;

  public:
    Moteur(int in1, int in2, int pwm) {
      pin_in1 = in1;
      pin_in2 = in2;
      pin_pwm = pwm;
      pinMode(in1, OUTPUT);
      pinMode(in2, OUTPUT);
      pinMode(pwm, OUTPUT);
    }

    void avancer(int vitesse) {
      digitalWrite(pin_in1, HIGH);
      digitalWrite(pin_in2, LOW);
      analogWrite(pin_pwm, vitesse); // 0–255
    }

    void reculer(int vitesse) {
      digitalWrite(pin_in1, LOW);
      digitalWrite(pin_in2, HIGH);
      analogWrite(pin_pwm, vitesse);
    }

    void arreter() {
      digitalWrite(pin_in1, LOW);
      digitalWrite(pin_in2, LOW);
      analogWrite(pin_pwm, 0);
    }
};
